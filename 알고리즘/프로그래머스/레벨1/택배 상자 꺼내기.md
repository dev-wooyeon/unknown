### 풀이 코드
``` java
class Solution {
    public int solution(int n, int w, int num) {
        
        int h = (int) Math.ceil((double) n / w); // 총 층수
        int[][] container = new int[h][w];
        int point = 0;
        int target_col_idx = 0;
        
        int box = 1;
        for (int i = 0; i < h; i++) {
            if (i % 2 == 0) { // 짝수층: 왼→오
                for (int j = 0; j < w && box <= n; j++) {
                    container[i][j] = box++;
                }
            }
            if (i % 2 != 0) { // 홀수층: 오→왼
                for (int j = w - 1; j >= 0 && box <= n; j--) {
                    container[i][j] = box++;
                }
            }
        }
        
        // num 위치 찾기
        int targetRow = 0;
        int targetCol = 0;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                if (container[i][j] == num) {
                    targetRow = i;
                    targetCol = j;
                }
            }
        }

        // target 위쪽 같은 column 박스 세기
        int answer = 0;
        for (int i = targetRow; i < h; i++) {
            if (container[i][targetCol] != 0) {
                answer++;
            }
        }

        return answer;
    }
    
    
}
/* 
 * 파라미터
 * n : 택바상자 넘버 1 ~ n
 * w : 택배 상자를 쌓을 수 있는 가로 길이
 * num : 꺼내려고 하는 택배 상자 번호
 *
 * 입출력 예시 분석
 * n(13) : 택바상자 넘버 1 ~ n
 * w(3) : 택배 상자를 쌓을 수 있는 가로 길이
 * num(6) : 꺼내려고 하는 택배 상자 번호
 * result(4) : 13, 12, 7, 6(꺼내려는 택배 상자번호 포함한 카운트)
 *
 * 자료구조
 * 2차원 배열
 */
```

class Solution {
    public int solution(int n, int w, int num) {
      int cnt = 0;
      while(num <= n) {
          num += (w - ((num-1) % w) -1) * 2 + 1;
          cnt++;
      }
  
      int answer = cnt;
      return answer;
  }
}
